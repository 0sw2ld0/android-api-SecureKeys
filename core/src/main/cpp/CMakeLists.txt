cmake_minimum_required(VERSION 3.5)

project(secure-keys)

link_directories(${JNI_LIBRARIES})

add_library(secure-keys

			crypto/sha256.cpp

			crypto/aes.h
			crypto/base64.h
			crypto/sha256.h
			crypto/crypto_wrapper.h)
